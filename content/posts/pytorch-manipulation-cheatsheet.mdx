## Broadcasting Rules

```python
# Same as NumPy: align right, dims compatible if equal or 1
a = torch.randn(3, 1, 5)
b = torch.randn(1, 4, 5)
c = a + b  # (3, 4, 5)

# Common patterns
x = torch.randn(32, 10)  # batch
w = torch.randn(10)      # weights
y = x + w                # (32, 10) - adds to each row

# Explicit broadcasting
x.unsqueeze(-1)  # add dim at end
x.unsqueeze(0)   # add dim at start
x[..., None]     # same as unsqueeze(-1)
```

## Tensor Creation

```python
# Zeros/Ones: specify shape
torch.zeros(3, 4)
torch.ones(2, 3, 4)
torch.full((3, 3), 7)  # fill with value

# Like existing tensor
torch.zeros_like(x)
torch.ones_like(x)

# Random
torch.randn(3, 4)      # normal(0, 1)
torch.rand(3, 4)       # uniform[0, 1)
torch.randint(0, 10, (3, 4))  # integers

# Sequences
torch.arange(0, 10, 2)  # [0, 2, 4, 6, 8]
torch.linspace(0, 1, 5) # [0, 0.25, 0.5, 0.75, 1]

# Identity
torch.eye(3)  # (3, 3) identity matrix
```

## Reshaping & Views

```python
# Reshape: (n,) → (m, k) where n=m*k
x.reshape(2, 3)
x.view(2, 3)     # must be contiguous
x.reshape(-1)    # flatten
x.reshape(-1, 4) # infer first dim

# Transpose
x.T              # (m, n) → (n, m)
x.transpose(0, 1)
x.permute(2, 0, 1)  # arbitrary reorder

# Squeeze/Unsqueeze
x.squeeze()      # remove dims of size 1
x.squeeze(1)     # remove specific dim if size 1
x.unsqueeze(0)   # add dim of size 1

# Flatten
x.flatten()      # → (n,)
x.flatten(1)     # flatten from dim 1 onwards
```

## Indexing & Slicing

```python
# Basic slicing (returns view)
x[0]           # first element
x[:, 1]        # second column
x[1:3, :]      # rows 1-2
x[..., -1]     # last element of last dim

# Boolean indexing (returns copy)
x[x > 0]       # elements > 0, flattened
x[x > 0] = 0   # set elements to 0

# Fancy indexing
indices = torch.tensor([0, 2, 4])
x[indices]     # select specific indices

# Advanced indexing
rows = torch.tensor([0, 1, 2])
cols = torch.tensor([1, 2, 0])
x[rows, cols]  # specific elements

# Masked selection
mask = x > 0
torch.masked_select(x, mask)  # flattened

# Index select
torch.index_select(x, dim=0, index=indices)
```

## Assignment & In-place Ops

```python
# Direct assignment
x[0] = 1
x[:, 0] = torch.ones(3)
x[x > 0] = 0

# In-place operations (save memory)
x.add_(1)      # x = x + 1
x.mul_(2)      # x = x * 2
x.clamp_(0, 1) # clip values
x.fill_(0)     # fill with value

# Scatter/Gather
x.scatter_(dim, index, src)
x.gather(dim, index)

# Advanced assignment
x.index_copy_(dim, index, tensor)
x.index_add_(dim, index, tensor)
x.index_fill_(dim, index, value)

# Masked fill
x.masked_fill_(mask, value)
x.masked_scatter_(mask, source)
```

## Reductions

```python
# Full reduction: (m, n) → ()
x.sum()
x.mean()
x.std()
x.var()
x.max()
x.min()
x.prod()

# Along dimension: (m, n) → (m,) or (n,)
x.sum(dim=0)   # column sums → (n,)
x.sum(dim=1)   # row sums → (m,)
x.sum(dim=1, keepdim=True)  # → (m, 1)

# Multiple dimensions
x.sum(dim=(0, 2))  # sum over dims 0 and 2

# With indices
x.max(dim=0)   # returns (values, indices)
x.argmax(dim=0)  # just indices

# Cumulative
x.cumsum(dim=0)
x.cumprod(dim=0)
```

## Matrix Operations

```python
# Dot product: (n,)·(n,) → ()
torch.dot(a, b)

# Matrix-vector: (m, n)·(n,) → (m,)
torch.mv(A, x)
A @ x

# Matrix-matrix: (m, n)·(n, k) → (m, k)
torch.mm(A, B)
A @ B

# Batch matrix multiply: (b, m, n)·(b, n, k) → (b, m, k)
torch.bmm(A, B)

# General matmul (handles broadcasting)
torch.matmul(A, B)
A @ B

# Outer product: (m,)×(n,) → (m, n)
torch.outer(a, b)

# Element-wise multiply
A * B
torch.mul(A, B)
```

## Stacking & Concatenation

```python
# Stack: creates new dimension

torch.stack([a, b], dim=1)

# Concatenate: along existing dimension

# Split
torch.split(x, 2, dim=0)  # chunks of size 2
torch.chunk(x, 3, dim=0)  # split into 3 parts

torch.unbind(x, dim=0)  # list of tensors
```

## Common ML Patterns

```python
# Normalize (z-score)
(x - x.mean(dim=0)) / x.std(dim=0)
```

## Avoiding Loops

```python
# BAD: Loop over batch
for i in range(batch_size):
    output[i] = model(x[i])

# GOOD: Vectorized
output = model(x)  # (batch_size, ...)

# BAD: Loop for element-wise op
for i in range(len(x)):
    y[i] = x[i] ** 2

# GOOD: Vectorized
y = x ** 2

# BAD: Loop for conditional
for i in range(len(x)):
    if x[i] > 0:
        y[i] = x[i]
    else:
        y[i] = 0

# GOOD: Use where or clamp
y = torch.where(x > 0, x, torch.zeros_like(x))
y = torch.clamp(x, min=0)  # ReLU

# BAD: Loop for gather
for i in range(len(indices)):
    result[i] = x[i, indices[i]]

# GOOD: Advanced indexing
result = x[torch.arange(len(indices)), indices]
```

## Performance Tips

1. **Vectorize everything** - avoid Python loops
2. **Use in-place ops** when possible (`add_`, `mul_`)
3. **Preallocate tensors** with `torch.empty()`
4. **Use views** over copies when possible
5. **Batch operations** - process multiple samples together
6. **Use `torch.no_grad()`** during inference
7. **Pin memory** for faster CPU→GPU transfer
8. **Use appropriate dtype** - `float16` for speed, `float32` for stability
9. **Profile your code** - `torch.profiler` or `torch.autograd.profiler`
10. **Fuse operations** - combine multiple ops when possible

## Advanced Broadcasting Examples

```python
# Batch matrix-vector multiply
# x: (batch, n, m), v: (m,) → (batch, n)
result = x @ v  # broadcasts v across batch

# Attention scores
# Q: (batch, seq_q, d), K: (batch, seq_k, d)
# → (batch, seq_q, seq_k)
scores = torch.einsum('bqd,bkd->bqk', Q, K) / math.sqrt(d)

# Batch outer product
# a: (batch, m), b: (batch, n) → (batch, m, n)
result = a.unsqueeze(2) * b.unsqueeze(1)
# or
result = torch.einsum('bi,bj->bij', a, b)

# Weighted sum
# x: (batch, seq, d), weights: (batch, seq)
# → (batch, d)
result = torch.einsum('bsd,bs->bd', x, weights)

# Pairwise operations with broadcasting
# x: (n, d), y: (m, d) → (n, m, d)
diff = x.unsqueeze(1) - y.unsqueeze(0)
# Then reduce: (n, m, d) → (n, m)
distances = diff.norm(dim=2)
```

## Practical Examples

### Attention Mechanism
```python
def attention(Q, K, V, mask=None):
    """
    Q: (batch, heads, seq_q, d_k)
    K: (batch, heads, seq_k, d_k)
    V: (batch, heads, seq_k, d_v)
    """
    d_k = Q.size(-1)
    scores = torch.matmul(Q, K.transpose(-2, -1)) / math.sqrt(d_k)
    
    if mask is not None:
        scores = scores.masked_fill(mask == 0, -1e9)
    
    attn = F.softmax(scores, dim=-1)
    return torch.matmul(attn, V), attn
```

### Batch Normalization
```python
def batch_norm(x, gamma, beta, eps=1e-5):
    """x: (batch, channels, height, width)"""
    # Compute mean and var over batch and spatial dims
    mean = x.mean(dim=(0, 2, 3), keepdim=True)
    var = x.var(dim=(0, 2, 3), keepdim=True, unbiased=False)
    
    # Normalize
    x_norm = (x - mean) / torch.sqrt(var + eps)
    
    # Scale and shift
    return gamma.view(1, -1, 1, 1) * x_norm + beta.view(1, -1, 1, 1)
```

### Efficient Pairwise Distance
```python
def pairwise_distance(x, y=None):
    """
    x: (n, d)
    y: (m, d) or None
    Returns: (n, m) or (n, n) if y is None
    """
    if y is None:
        y = x
    
    # ||x - y||^2 = ||x||^2 + ||y||^2 - 2<x, y>
    x_norm = (x ** 2).sum(1).view(-1, 1)
    y_norm = (y ** 2).sum(1).view(1, -1)
    dist = x_norm + y_norm - 2.0 * torch.mm(x, y.T)
    
    return torch.clamp(dist, min=0.0).sqrt()
```
